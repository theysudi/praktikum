<?php
    class Mobil
    {
        public $mobil = 'Xenia';

        public function setName($nama)
        {
            $this->mobil = $nama;
        }

        public function getName()
        {
            return $this->mobil;
        }
    }

    $mobil = new Mobil;
    $mobil->setName('Avanza');
    echo $mobil->getName();

<?php
    class Mobil
    {
        public $mobil = 'Xenia';

         function __construct()
